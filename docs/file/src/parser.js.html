    const fs = require('fs');

class Func
{
  constructor (name, args, commands=[])
  {
    this.name = name;
    this.args = args;
    this.commands = commands;
  }
}

class Try
{
  constructor (tryblk=[], catchblk=[], finallyblk=[])
  {
    this.try = tryblk;
    this.catch = catchblk;
    this.finally = finallyblk;
  }
}

module.exports = class Parser
{
  constructor ()
  {}

  _parseFunc (idx, commands)
  {
    const def = commands[idx++].split(/\s+/);
    def.shift();
    const name = def.shift();
    const args = def;
    const funCmds = [];
    let cmd;
    while (idx < commands.length) {
      [idx, cmd] = this._parseCommand(idx, commands);
      if (cmd == 'enuf') {
        return [idx, new Func(name, args, funCmds)];
      }
      funCmds.push(cmd);
      idx++;
    }
    throw new SyntaxError(`Invalid func definition '${name}'`);
  }

  _parseTry (idx, commands)
  {
    const cmds = [[], [], []];
    let state = 0;
    let cmd;
    idx++;
    while (idx < commands.length) {
      [idx, cmd] = this._parseCommand(idx, commands);
      if (cmd == 'catch') {
        if (state > 0) throw new SyntaxError(`Unexpected token 'catch'`);
        state = 1;
      }
      else if (cmd == 'finally') {
        if (state > 1) throw new SyntaxError(`Unexpected token 'finally'`);
        state = 2;
      }
      else if (cmd == 'yrt') {
        if (state == 0) throw new SyntaxError(`Unexpected token 'yrt'`);
        return [idx, new Try(...cmds)];
      }
      else {
        cmds[state].push(cmd);
      }
      idx++;
    }
  }

  _parseCommand (idx, commands)
  {
    const cmd = commands[idx];

    if (/^func\s/.test(cmd)) {
      return this._parseFunc(idx, commands);
    }
    if (/^try$/.test(cmd)) {
      return this._parseTry(idx, commands);
    }

    return [idx, cmd];
  }

  parse (data)
  {
    const result = {};
    const commands = data.toString()
      .split(/\n|;/)
      .map((cmd) => cmd.trim())
      .filter((cmd) => cmd.length);

    let idx = 0, cmd;

    while (idx < commands.length) {
      [idx, cmd] = this._parseCommand(idx, commands);
      if (cmd instanceof Func) {
        result[cmd.name] = cmd;
      }
      idx++;
    }

    return result;
  }

  parseFile (filename)
  {
    const data = fs.readFileSync(filename);
    return this.parse(data);
  }
}
